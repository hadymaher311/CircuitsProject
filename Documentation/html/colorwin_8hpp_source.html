62  // Example usage: std::cout << color(yellow) << "This is a warning color!\n";
63  class color
64  {
65  public:
66  color(CW_COLORS color) : m_color(color), m_console_handle(INVALID_HANDLE_VALUE)
67  {
68  CONSOLE_SCREEN_BUFFER_INFO console_info;
69  m_console_handle = GetStdHandle(STD_OUTPUT_HANDLE);
70  if (!GetConsoleScreenBufferInfo(m_console_handle, &console_info))
71  {
72  m_console_handle = GetStdHandle(STD_ERROR_HANDLE);
73  if (!GetConsoleScreenBufferInfo(m_console_handle, &console_info)) // maybe standard output device has been redirected, try the standard error device
74  {
75  m_console_handle = INVALID_HANDLE_VALUE;
76  }
77  }
78  }
79 
80  ~color()
81  {
82  if (m_console_handle != INVALID_HANDLE_VALUE)
83  {
84  // Restore the previous color.
85  SetConsoleTextAttribute(m_console_handle, get_color_stack().top());
86  get_color_stack().pop();
87  }
88  }
89 
90  private:
91  void change_color() const
92  {
93  if (m_console_handle == INVALID_HANDLE_VALUE)
94  return; // Can't get console info, can't change color.
95  CONSOLE_SCREEN_BUFFER_INFO console_info;
96  GetConsoleScreenBufferInfo(m_console_handle, &console_info);
97  // save the current attributes for restoration on destruction.
98  get_color_stack().push(console_info.wAttributes);
99  SetConsoleTextAttribute(m_console_handle, 0x0F & m_color | 0xf0 & console_info.wAttributes); // save the background color
100  }
